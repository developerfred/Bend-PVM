558    /// Mutate a single input
559    fn mutate_input(&mut self, input: &mut TestInput) {
560        match input {
561            TestInput::Integer(val) => {
562                if self.rng.gen::<f64>() < 0.5 {
563                    *val = self.rng.gen_range(-1000..1000);
564                } else {
565                    *val = val.wrapping_add(self.rng.gen_range(-10..=10));
566                }
567            }
568            TestInput::String(s) => {
569                if !s.is_empty() && self.rng.gen::<f64>() < 0.3 {
570                    let idx = self.rng.gen_range(0..s.len());
571                    s.remove(idx);
572                } else if s.len() < 100 && self.rng.gen::<f64>() < 0.3 {
573                    s.push(self.rng.gen::<char>());
574                } else {
575                    *s = (0..self.rng.gen_range(1..20)).map(|_| {
576                        self.rng.gen_range(b'a'..=b'z') as char
577                    }).collect();
578                }
579            }
580            TestInput::Boolean(val) => {
581                *val = !*val;
582            }
583            TestInput::Address(addr) => {
584                if !addr.is_empty() {
585                    let idx = self.rng.gen_range(0..addr.len());
586                    addr[idx] = self.rng.gen();
587                }
588            }
589            _ => {
590                // For other types, regenerate randomly
591                *input = match self.rng.gen_range(0..4) {
592                    0 => TestInput::Integer(self.rng.gen()),
593                    1 => TestInput::String((0..10).map(|_| self.rng.gen::<char>()).collect()),
594                    2 => TestInput::Boolean(self.rng.gen()),
595                    _ => TestInput::Integer(42),
596                };
597            }
598        }
599    }

662    /// Get fuzzing statistics
663    pub fn get_stats(&self) -> FuzzingStats {
664        FuzzingStats {
665            total_executions: self.execution_history.iter().map(|r| r.total_tests).sum(),
666            success_rate: if !self.execution_history.is_empty() {
667                let total: u32 = self.execution_history.iter().map(|r| r.total_tests).sum();
668                let successful: u32 = self.execution_history.iter().map(|r| r.passed_tests).sum();
669                if total > 0 { (successful as f64 / total as f64) * 100.0 } else { 0.0 }
670            } else { 0.0 },
671            corpus_size: self.corpus.len(),
672            coverage_points: self.coverage_map.len(),
673            unique_errors: self.execution_history.iter()
674                .flat_map(|r| r.error_counts.keys())
675                .collect::<HashSet<_>>()
676                .len(),
677            avg_execution_time: if !self.execution_history.is_empty() {
678                self.execution_history.iter().map(|r| r.execution_time_ms as f64).sum::<f64>() / self.execution_history.len() as f64
679            } else { 0.0 },
680        }
681    }
