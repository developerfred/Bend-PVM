5/// Represents the ABI for a contract

7pub struct ContractABI {

11    /// Contract version

179/// Generate an ABI from contract metadata
180pub fn generate_abi(metadata: &ContractMetadata) -> ContractABI {
181    let mut methods = Vec::new();
182    
183    // Add methods from the metadata
184    for (name, function) in &metadata.functions {
185        methods.push(function_to_method_abi(name, function));
186    }
187    
188    // For this example, we're not implementing events, errors, state variables, or types
189    
190    ContractABI {
191        name: metadata.name.clone(),
192        version: metadata.version.clone(),
193        methods,
194        events: Vec::new(),
195        errors: Vec::new(),
196        state_variables: Vec::new(),
197        types: Vec::new(),
198    }
199}

201/// Convert a function metadata to a method ABI
202fn function_to_method_abi(name: &str, function: &FunctionMetadata) -> MethodABI {
203    // Convert function parameters to ABI parameters
204    let mut inputs = Vec::new();
205    for param in &function.params {
206        inputs.push(ParameterABI {
207            name: param.name.clone(),
208            type_: param.type_name.clone(),
209            components: None,
210            indexed: None,
211        });
212    }
213    
214    // Convert function return type to ABI parameters
215    let outputs = if let Some(return_type) = &function.return_type {
216        vec![ParameterABI {
217            name: "".to_string(),
218            type_: return_type.clone(),
219            components: None,
220            indexed: None,
221        }]
222    } else {
223        Vec::new()
224    };
225    
226    // Convert function selector to hex string
227    let selector = hex::encode(function.selector);
228    
229    MethodABI {
230        name: name.to_string(),
231        selector: format!("0x{}", selector),
232        type_: MethodType::Function,
233        inputs,
234        outputs,
235        state_mutability: StateMutability::NonPayable, // Default to non-payable
236        payable: false,
237    }
238}
