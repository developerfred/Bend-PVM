5/// Eta reduction optimization pass
6///
7/// This pass performs eta reduction, which is a transformation that simplifies
8/// functions that just wrap other functions without adding functionality.
9/// For example, `lambda x: f(x)` can be simplified to just `f`.
10pub struct EtaReductionPass;
11
12impl EtaReductionPass {
13    /// Creates a new eta reduction pass
14    pub fn new() -> Self {
15        EtaReductionPass
16    }

18    /// Optimize a block using eta reduction
19    fn optimize_block(&self, block: &Block) -> (Block, bool) {
20        let mut new_statements = Vec::new();
21        let mut modified = false;
22        
23        for statement in &block.statements {
24            let (optimized_statement, stmt_modified) = self.optimize_statement(statement);
25            new_statements.push(optimized_statement);
26            modified = modified || stmt_modified;
27        }
28        
29        (
30            Block {
31                statements: new_statements,
32                location: block.location.clone(),
33            },
34            modified
35        )
36    }

38    /// Optimize a statement using eta reduction
39    fn optimize_statement(&self, statement: &Statement) -> (Statement, bool) {
40        match statement {
41            Statement::Assignment { pattern, value, location } => {
42                let (optimized_expr, expr_modified) = self.optimize_expr(value);
43                
44                (
45                    Statement::Assignment {
46                        pattern: pattern.clone(),
47                        value: optimized_expr,
48                        location: location.clone(),
49                    },
50                    expr_modified
51                )
52            },
53            Statement::Return { value, location } => {
54                let (optimized_expr, expr_modified) = self.optimize_expr(value);
55                
56                (
57                    Statement::Return {
58                        value: optimized_expr,
59                        location: location.clone(),
60                    },
61                    expr_modified
62                )
63            },
64            Statement::If { condition, then_branch, else_branch, location } => {
65                let (optimized_condition, condition_modified) = self.optimize_expr(condition);
66                let (optimized_then, then_modified) = self.optimize_block(then_branch);
67                let (optimized_else, else_modified) = self.optimize_block(else_branch);
68                
69                let modified = condition_modified || then_modified || else_modified;
70                
71                (
72                    Statement::If {
73                        condition: optimized_condition,
74                        then_branch: optimized_then,
75                        else_branch: optimized_else,
76                        location: location.clone(),
77                    },
78                    modified
79                )
80            },
81            Statement::Expr { expr, location } => {
82                let (optimized_expr, expr_modified) = self.optimize_expr(expr);
83                
84                (
85                    Statement::Expr {
86                        expr: optimized_expr,
87                        location: location.clone(),
88                    },
89                    expr_modified
90                )
91            },
92            // For other statement types, implement optimization logic
93            // This is a simplified implementation
94            _ => (statement.clone(), false),
95        }
96    }

98    /// Optimize an expression using eta reduction
99    fn optimize_expr(&self, expr: &Expr) -> (Expr, bool) {
100        match expr {
101            Expr::Lambda { params, body, location } => {
102                // First, recursively optimize the body
103                let (optimized_body, body_modified) = self.optimize_expr(body);
104                
105                // Check for eta-reducible form: lambda x: f(x)
106                if let Expr::FunctionCall { function, args, named_args, .. } = &optimized_body {
107                    // Check if this is a simple function call that can be eta-reduced
108                    if named_args.is_empty() && args.len() == params.len() {
109                        // Check that each argument is a parameter variable in the same order
110                        let mut can_eta_reduce = true;
111                        let mut param_names = HashSet::new();
112                        
113                        // Collect parameter names
114                        for param in params.iter() {
115                            param_names.insert(&param.name);
116                        }
117                        
118                        // Check that each argument is a parameter variable
119                        for (i, arg) in args.iter().enumerate() {
120                            if let Expr::Variable { name, .. } = arg {
121                                if !param_names.contains(name) || &params[i].name != name {
122                                    can_eta_reduce = false;
123                                    break;
124                                }
125                            } else {
126                                can_eta_reduce = false;
127                                break;
128                            }
129                        }
130                        
131                        if can_eta_reduce {
132                            // Eta-reduce the lambda to the function
133                            return ((**function).clone(), true);
134                        }
135                    }
136                }
137                
138                // If not eta-reducible, return the optimized lambda
139                (
140                    Expr::Lambda {
141                        params: params.clone(),
142                        body: Box::new(optimized_body),
143                        location: location.clone(),
144                    },
145                    body_modified
146                )
147            },
148            Expr::UnsccopedLambda { params, body, location } => {
149                // Recursively optimize the body
150                let (optimized_body, body_modified) = self.optimize_expr(body);
151                
152                // Check for eta-reducible form: lambda x: f(x)
153                if let Expr::FunctionCall { function, args, named_args, .. } = &optimized_body {
154                    // Check if this is a simple function call that can be eta-reduced
155                    if named_args.is_empty() && args.len() == params.len() {
156                        // Check that each argument is a parameter variable in the same order
157                        let mut can_eta_reduce = true;
158                        let param_set: HashSet<&String> = params.iter().collect();
159                        
160                        // Check that each argument is a parameter variable
161                        for (i, arg) in args.iter().enumerate() {
162                            if let Expr::Variable { name, .. } = arg {
163                                if !param_set.contains(name) || &params[i] != name {
164                                    can_eta_reduce = false;
165                                    break;
166                                }
167                            } else {
168                                can_eta_reduce = false;
169                                break;
170                            }
171                        }
172                        
173                        if can_eta_reduce {
174                            // Eta-reduce the lambda to the function
175                            return ((**function).clone(), true);
176                        }
177                    }
178                }
179                
180                // If not eta-reducible, return the optimized lambda
181                (
182                    Expr::UnsccopedLambda {
183                        params: params.clone(),
184                        body: Box::new(optimized_body),
185                        location: location.clone(),
186                    },
187                    body_modified
188                )
189            },
190            Expr::BinaryOp { left, operator, right, location } => {
191                let (optimized_left, left_modified) = self.optimize_expr(left);
192                let (optimized_right, right_modified) = self.optimize_expr(right);
193                
194                let modified = left_modified || right_modified;
195                
196                (
197                    Expr::BinaryOp {
198                        left: Box::new(optimized_left),
199                        operator: operator.clone(),
200                        right: Box::new(optimized_right),
201                        location: location.clone(),
202                    },
203                    modified
204                )
205            },
206            Expr::FunctionCall { function, args, named_args, location } => {
207                let (optimized_function, function_modified) = self.optimize_expr(function);
208                
209                let mut optimized_args = Vec::new();
210                let mut args_modified = false;
211                
212                for arg in args {
213                    let (optimized_arg, arg_modified) = self.optimize_expr(arg);
214                    optimized_args.push(optimized_arg);
215                    args_modified = args_modified || arg_modified;
216                }
217                
218                let mut optimized_named_args = named_args.clone();
219                let mut named_args_modified = false;
220                
221                for (name, arg) in named_args {
222                    let (optimized_arg, arg_modified) = self.optimize_expr(arg);
223                    if arg_modified {
224                        optimized_named_args.insert(name.clone(), optimized_arg);
225                        named_args_modified = true;
226                    }
227                }
228                
229                let modified = function_modified || args_modified || named_args_modified;
230                
231                (
232                    Expr::FunctionCall {
233                        function: Box::new(optimized_function),
234                        args: optimized_args,
235                        named_args: optimized_named_args,
236                        location: location.clone(),
237                    },
238                    modified
239                )
240            },
241            Expr::Block { block, location } => {
242                // Optimize the block
243                let (optimized_block, block_modified) = self.optimize_block(block);
244                
245                (
246                    Expr::Block {
247                        block: optimized_block,
248                        location: location.clone(),
249                    },
250                    block_modified
251                )
252            },
253            // For other expression types, no optimization needed
254            _ => (expr.clone(), false),
255        }
256    }
257}

259impl OptimizationPass for EtaReductionPass {
260    fn name(&self) -> &'static str {
261        "eta_reduction"
262    }
263    
264    fn description(&self) -> &'static str {
265        "Performs eta reduction for functions that just wrap other functions"
266    }
267    
268    fn run(&self, program: Program) -> Result<OptimizationResult, OptimizationError> {
269        let mut modified = false;
270        let mut new_definitions = Vec::new();
271        
272        // Optimize each definition
273        for definition in &program.definitions {
274            match definition {
275                Definition::FunctionDef { name, params, return_type, body, checked, location } => {
276                    // Optimize the function body
277                    let (optimized_body, body_modified) = self.optimize_block(body);
278                    modified = modified || body_modified;
279                    
280                    new_definitions.push(Definition::FunctionDef {
281                        name: name.clone(),
282                        params: params.clone(),
283                        return_type: return_type.clone(),
284                        body: optimized_body,
285                        checked: *checked,
286                        location: location.clone(),
287                    });
288                },
289                // Other definition types don't need optimization
290                _ => new_definitions.push(definition.clone()),
291            }
292        }
293        
294        // Return the result
295        if modified {
296            Ok(OptimizationResult::Modified(Program {
297                imports: program.imports.clone(),
298                definitions: new_definitions,
299                location: program.location.clone(),
300            }))
301        } else {
302            Ok(OptimizationResult::Unchanged(program))
303        }
304    }
305}
