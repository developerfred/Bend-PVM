3/// Represents a token in the Bend-PVM language
4#[derive(Debug, Clone, PartialEq)]
5pub enum Token {
6    // Keywords
7    Def,
8    Type,
9    Object,
10    Return,
11    If,
12    Else,
13    Match,
14    Case,
15    Fold,
16    Bend,
17    When,
18    Fork,
19    Open,
20    With,
21    Use,
22    Lambda,
23    In,
24    Let,
25    Switch,
26    Import,
27    From,
28    As,
29
30    // Symbols
31    LParen,
32    RParen,
33    LBrace,
34    RBrace,
35    LBracket,
36    RBracket,
37    Colon,
38    Semicolon,
39    Comma,
40    Dot,
41    Arrow,
42    FatArrow,
43    Equal,
44    Tilde,
45    BackTick,
46    
47    // Operators
48    Plus,
49    Minus,
50    Star,
51    Slash,
52    Percent,
53    Caret,
54    Ampersand,
55    Pipe,
56    GreaterThan,
57    LessThan,
58    GreaterEqual,
59    LessEqual,
60    EqualEqual,
61    NotEqual,
62    PlusEqual,
63    MinusEqual,
64    StarEqual,
65    SlashEqual,
66    PercentEqual,
67    CaretEqual,
68    AmpersandEqual,
69    PipeEqual,
70    
71    // Literals
72    Identifier(String),
73    UintLiteral(u32),  // For u24
74    IntLiteral(i32),   // For i24
75    FloatLiteral(f32), // For f24
76    StringLiteral(String),
77    CharLiteral(char),
78    SymbolLiteral(String),
79

81    EOF,
